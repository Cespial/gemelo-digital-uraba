\begin{frame}
\frametitle{Hoy cada finca despacha por su cuenta}
\tagline{rojo}{El Problema}

% 5. ¿CUÁNTO CUESTA EL HUB POR CAJA?
% ============================================================
\begin{frame}
\frametitle{¿Cuánto cuesta operar el hub por caja?}
\tagline{azul}{Cada componente, paso a paso}

\vspace{2pt}
{\small El hub tiene tres costos por caja y un costo fijo anual (OPEX):}

\vspace{6pt}
\paso{1} {\small \textbf{Transporte hub $\rightarrow$ Puerto Antioquia}: Centro + Mini-Hub consolidan y despachan
12 mulas/día llenas. Costo ponderado: \textbf{\$582/caja}. Las 4 fincas Zungo ya exportan desde el embarcadero.}

{\tiny\color{gris}\hspace{24pt}Centro: 10 mulas $\times$ \$720K = \$7.2M/día · Mini-Hub: 2 $\times$ \$600K = \$1.2M · Total: \$8.4M / 14,440 cajas PA/día. Zungo: sin cambio.}

\vspace{4pt}
\paso{2} {\small \textbf{Consolidación} (paletizar, inspeccionar, cargar): \textbf{\$42/caja}.}

\vspace{4pt}
\paso{3} {\small \textbf{Transporte finca $\rightarrow$ hub} (vehículos propios, trayecto corto): \textbf{\$80/caja} (supuesto, por validar).}

\vspace{6pt}
\begin{center}
    \colorbox{verdeclaro}{\parbox{10cm}{\centering\small\vspace{4pt}
        \textbf{Hub/caja (PA):} \$582 + \$42 + \$80 $=$ \textcolor{verde}{\textbf{\$704}} {\scriptsize~(ponderado G20: \$617)}
    \vspace{4pt}}}
\end{center}

\vspace{4pt}
\paso{4} {\scriptsize \textbf{OPEX}: 6 personas \$215M + servicios \$73M + mant./seguros \$38M + admin \$36M $=$ \textbf{\$362M/año} (\$69/caja).}

{\tiny\color{gris}\hspace{24pt}CAPEX terreno: \$0 (Chinita propio). Construcción: \$800--1,200M.}
\end{frame}

\tagline{verde}{Cada número tiene su origen}

\vspace{2pt}
\begin{center}
\small
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{l r l}
    \toprule
    \textcolor{rojo}{Costo sin hub (\$/viaje)} & \textcolor{rojo}{\$750/caja} & {\scriptsize 26 viajes, 76\% llenado (diap.\ 3)} \\
    \textcolor{verde}{Costo con hub}           & \textcolor{verde}{\$617/caja} & {\scriptsize \$713 + \$42 + \$80 (diap.\ 5)} \\
    \midrule
    \textbf{Ahorro por caja}                   & \textbf{\$133} & \\
    $\times$ 5.25M cajas/año                   & \textbf{\$699M} & {\scriptsize bruto} \\
    $-$ OPEX hub                               & --\$362M       & {\scriptsize 6 pers.\ + servicios} \\
    \midrule
    \textbf{Ahorro neto}                       & \textcolor{verde}{\textbf{\$337M/año}} & \\
    \addlinespace
    CAPEX \$800M $\div$ \$337M                 & \textcolor{verde}{\textbf{2.4 años}} & {\scriptsize payback} \\
    CAPEX \$1,200M $\div$ \$337M               & \textcolor{azul}{\textbf{3.6 años}} & {\scriptsize payback} \\
    \bottomrule
\end{tabular}
\end{center}

\vspace{2pt}
\colorbox{verdeclaro}{\parbox{\dimexpr\textwidth-12pt}{\scriptsize\strut
\textcolor{verde}{\textbf{Trazabilidad:}} \$750 $=$ 15 fincas PA (\$890) + 4 Zungo (\$398), ponderado.
\$617 $=$ 12 mulas PA consolidadas (\$704/caja) + 4 Zungo sin cambio, ponderado.
\$337M $=$ (\$750--\$617) $\times$ 5.25M -- \$362M OPEX. Terreno \$0.\strut}}

\vspace{1pt}
{\tiny\color{gris} Supuesto: modelo \$/viaje adoptado. Bajo \$/caja actual: ahorro $=$ \$0. Fuente: Anderson Mestra, Mar 2026.}
