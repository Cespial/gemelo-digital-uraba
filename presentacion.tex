\textbf{Costo anual SC2: \$5,058M} (vs \$2,133M hoy). \textcolor{rojo}{\textbf{+137\%.}}

\begin{center}
\scriptsize
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{r l r l}

    & \textbf{Componente} & \textbf{Costo} & \textbf{Cálculo} \\
    \midrule
    \textcolor{azul}{\textbf{1.}} & Hub $\rightarrow$ PA (mulas llenas) & \$568/caja & {\tiny 10 mulas $\times$ \$720K $\div$ 12,674 cajas/día} \\
    \textcolor{azul}{\textbf{2.}} & Consolidación en hub & \$42/caja & {\tiny Paletizar + inspeccionar + cargar} \\
    \textcolor{azul}{\textbf{3.}} & Finca $\rightarrow$ hub & \$80/caja & {\tiny Vehículos propios, distancia corta (por validar)} \\
    \midrule
    & \textcolor{verde}{\textbf{Total hub}} & \textcolor{verde}{\textbf{\$690/caja}} & \\
    \bottomrule
\end{tabular}
\end{center}

\vspace{4pt}
\colorbox{rojoclaro}{\scriptsize\strut\hspace{4pt}\textcolor{rojo}{\textbf{Sin hub (SC2):}} 12 fincas Centro pagan \$919/caja promedio (16 viajes parciales)\hspace{4pt}}

\vspace{4pt}
{\scriptsize\textbf{Ahorro por caja:} \$919 $-$ \$690 $=$ \textcolor{verde}{\textbf{\$229/caja}} para cada caja que pase por el Centro.}

\vspace{2pt}
{\tiny\color{gris} Las 3 fincas Mini-Hub (\$681/caja) y 4 Zungo (\$324/caja) no pasan por el hub. Mula: \$300K $\times$ 2.4 (24 pallets).}
