% 5. DÓNDE SE AHORRAN LOS VIAJES (DESAGREGADO)
% ============================================================
\begin{frame}
\frametitle{\textquestiondown De d\'onde salen los 6 viajes menos?}
\tagline{verde}{La l\'ogica de la consolidaci\'on}

\vspace{1pt}
{\scriptsize \textbf{Sin hub:} cada finca redondea ARRIBA. Si produce 9.4 pallets, necesita 1 viaje de 12
$\rightarrow$ \textbf{2.6 pallets vac\'ios}. Multiplicado por 12 fincas $=$ mucho espacio desperdiciado.}

\vspace{3pt}
\begin{center}
\scriptsize
\renewcommand{\arraystretch}{0.8}
\begin{tabular}{l r r r}
    \toprule
    \textbf{Finca Centro} & \textbf{Pall/d\'ia} & \textbf{Viajes ($\div$12)} & \textbf{Pall.\ vac\'ios} \\
    \midrule
    Chinita       & 37.5 & 4 & 10.5 \\
    Tucanes       & 30.3 & 3 & 5.7 \\
    Maria Bonita  & 29.0 & 3 & 7.0 \\
    Sta.\ Marta   & 28.6 & 3 & 7.4 \\
    Sta.\ Helena  & 12.4 & 2 & \textcolor{rojo}{11.6} \\
    Llave         & 15.2 & 2 & 8.8 \\
    {\color{gris}\scriptsize ... 6 m\'as} & & & \\
    \midrule
    \textbf{Total 12} & \textbf{234.8} & \textbf{26} & \textcolor{rojo}{\textbf{77.2 vac\'ios}} \\
    \bottomrule
\end{tabular}
\end{center}

\vspace{2pt}
\begin{block}{Con hub: se redondea UNA sola vez}
\scriptsize
234.8 pallets $\div$ 12 $=$ 19.6 $\rightarrow$ \textbf{20 viajes} (98\% llenado).
Los pallets sobrantes de Chinita llenan el espacio de Maritu.
\textbf{Se eliminan 6 viajes de ``espacio vac\'io''.}
Los 77 pallets desperdiciados se convierten en 2.4 ($=$ 234.8 $-$ 20$\times$12).
\end{block}
{\tiny\color{gris} Mini-Hub (4 viajes) y Zungo (11 viajes) no cambian. Total G20: 41$\rightarrow$35 viajes.}
\end{frame}
