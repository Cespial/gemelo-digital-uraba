% 7. LA LÓGICA DE LA CONSOLIDACIÓN
% ============================================================
\begin{frame}
\frametitle{\textquestiondown Por qu\'e 10 mulas en vez de 26 sencillos?}
\tagline{verde}{La L\'ogica de la Consolidaci\'on}

\vspace{1pt}
{\scriptsize \textbf{Sin hub:} cada finca redondea ARRIBA. Si produce 9.4 pallets, necesita 1 sencillo de 12 $\rightarrow$ 2.6 pallets vac\'ios.}

\vspace{3pt}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
{\scriptsize\color{rojo}\textbf{SIN HUB: 26 sencillos (12 pallets c/u)}}
\vspace{1pt}

\scriptsize
\renewcommand{\arraystretch}{0.8}
\begin{tabular}{l r r r}
    \toprule
    \textbf{Finca} & \textbf{Pall/d} & \textbf{Viajes} & \textbf{Vac\'ios} \\
    \midrule
    Chinita       & 37.5 & 4 & 10.5 \\
    Tucanes       & 30.3 & 3 & 5.7 \\
    Maria Bonita  & 29.0 & 3 & 7.0 \\
    Sta.\ Helena  & 12.4 & 2 & \textcolor{rojo}{11.6} \\
    Maritu        &  9.4 & 1 & 2.6 \\
    {\color{gris}\tiny ... 7 m\'as} & & & \\
    \midrule
    \textbf{12 Centro} & \textbf{234.8} & \textcolor{rojo}{\textbf{26}} & \textcolor{rojo}{\textbf{77.2}} \\
    \bottomrule
\end{tabular}

\vspace{2pt}
{\tiny 26 viajes $\times$ 12p $=$ 312 slots. Solo 234.8 llenos.\\
\textcolor{rojo}{\textbf{77.2 pallets de espacio desperdiciado.}}}
\end{column}

\begin{column}{0.48\textwidth}
\begin{block}{\scriptsize Con hub: se consolida y se manda en mulas}
\scriptsize
234.8 pallets $\div$ 24 (mula) $=$ 9.78 $\rightarrow$ \textbf{10 mulas}

10 mulas $\times$ 24p $=$ 240 slots $\rightarrow$ \textbf{solo 5.2 vac\'ios}

\vspace{3pt}
\textcolor{verde}{\textbf{De 77.2 pallets desperdiciados a 5.2.}}

Los pallets sobrantes de una finca llenan el espacio de otra. Y la mula carga el \textbf{doble} que un sencillo.
\end{block}

\vspace{2pt}
\colorbox{verdeclaro}{\parbox{\dimexpr\linewidth-8pt}{\tiny\strut
\textbf{26 sencillos $\rightarrow$ 10 mulas}\\
16 viajes menos en la v\'ia.\\
Mini-Hub (4) y Zungo (11) no cambian.\\
\textbf{Total G20: 41 $\rightarrow$ 25.}\strut}}
\end{column}
\end{columns}
\end{frame}
